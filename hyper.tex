\begin{mathpar}
\inferrule{j ∈ I \\ Γ ⊢_I a : T \\Γ ⊢_I p : \PE i j T · a}{Γ ⊢_{I,i} (a,_i p) : T}

\inferrule{j ∈ I \\ Γ ⊢_{I,i} a : T}{Γ ⊢_I \PE i j a : \PE i j T ‌· a (ij)}
\end{mathpar}


\begin{align*}
  \PE i j U · T &= T(ij) → U \\
  \PE i j {(A → U)} · P &= \PI j {A(ij)} P  → U \\
  \PE i j {((x:A) → B x)} · f &= (x:A(ij)) → \PE i j {(B x(ji))} · f x \\
  \PE i j {(F · a)} · p &= \PE i j F · \PI j {a(ij)} p \\
  \PE i j {(λx. b[x])} · u & & \text{not well typed} \\
  \PE i j {(λx. b[x])} u &= \PE i j {(b [u(ij)])} \\
  \PE i j {\PI i a p}  &= p \\
  \PE i j {\PI k a p}  &= \PI k {\PE i j a} {\PE i j p} & k ≠ i \\
\end{align*}
